$tls12Client = "HKLM:\SYSTEM\CurrentControlSet\Control\SecurityProviders\SCHANNEL\Protocols\TLS 1.2\Client"
New-Item $tls12Client -Force | Out-Null
New-ItemProperty $tls12Client -Name "Enabled"            -Value 1 -PropertyType DWORD -Force | Out-Null
New-ItemProperty $tls12Client -Name "DisabledByDefault"  -Value 0 -PropertyType DWORD -Force | Out-Null



$tls12Server = "HKLM:\SYSTEM\CurrentControlSet\Control\SecurityProviders\SCHANNEL\Protocols\TLS 1.2\Server"
New-Item $tls12Server -Force | Out-Null
New-ItemProperty $tls12Server -Name "Enabled"            -Value 1 -PropertyType DWORD -Force | Out-Null
New-ItemProperty $tls12Server -Name "DisabledByDefault"  -Value 0 -PropertyType DWORD -Force | Out-Null


$dotnet64 = "HKLM:\SOFTWARE\Microsoft\.NETFramework\v4.0.30319"
$dotnet32 = "HKLM:\SOFTWARE\Wow6432Node\Microsoft\.NETFramework\v4.0.30319"
New-ItemProperty $dotnet64 -Name "SchUseStrongCrypto" -Value 1 -PropertyType DWORD -Force | Out-Null
New-ItemProperty $dotnet32 -Name "SchUseStrongCrypto" -Value 1 -PropertyType DWORD -Force | Out-Null

Write-Host "TLS 1.2 yapılandırması tamamlandı. Sistem yeniden başlatılmalıdır." -ForegroundColor Green
